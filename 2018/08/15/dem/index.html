def word_cloud(self, text):
    words_list = []
    word_generator = jieba.cut(text, cut_all=False)  # 返回的是一个迭代器
    for word in word_generator:
        if len(word) > 1:  # 去掉单字
             words_list.append(word)
    return ' '.join(words_list)
